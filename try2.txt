function filterUsers() {
    const input = document.getElementById('searchUserInput');
    const filter = input.value.toLowerCase();
    const userList = document.getElementById("userList");
    const checkboxes = userList.getElementsByClassName('form-check');

    // Rendre visible la liste si un filtre est appliquÃ©
    userList.style.display = filter.length > 0 ? 'block' : 'none';

    Array.from(checkboxes).forEach(function (checkbox) {
        const label = checkbox.querySelector('label');
        const txtValue = label ? label.textContent || label.innerText : '';
        checkbox.style.display = txtValue.toLowerCase().includes(filter) ? '' : 'none';
    });
}


public IActionResult Create()
{
